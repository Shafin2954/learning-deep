## The task

Given a graph at time $t$, predict which edges will appear at time $t+1$ — or equivalently,
which pairs of nodes should be connected that aren't yet. Applications: friend recommendation,
knowledge graph completion, drug-target interaction prediction.

The standard formulation: learn a scoring function $s: V \times V \rightarrow \mathbb{R}$ such
that $s(u,v) > s(u,w)$ whenever $(u,v)$ is a true edge and $(u,w)$ is not.

## Scoring functions

Given node embeddings $h_u, h_v \in \mathbb{R}^d$:

| Scorer | Formula | Parameters |
|--------|---------|-----------|
| Dot product | $h_u^T h_v$ | 0 |
| Hadamard | $\text{MLP}(h_u \odot h_v)$ | MLP weights |
| Bilinear | $h_u^T W h_v$ | $W \in \mathbb{R}^{d \times d}$ |
| Concatenation | $\text{MLP}([h_u \| h_v])$ | MLP weights |

The dot product is the most parameter-efficient and works well in practice.

## Negative sampling

You have positive edges (real connections). You need negative examples (non-edges) for binary
cross-entropy training. **Uniform random negative sampling**: for each positive edge $(u,v)$,
sample $K$ random nodes $w$ and treat $(u,w)$ as negative (assuming the graph is sparse enough
that randomly sampled pairs are unlikely to be true edges).

## Building the link predictor

```python
from torch_geometric.nn import GCNConv

class LinkPredictor(nn.Module):
    def __init__(self, in_ch, hidden):
        super().__init__()
        # GNN encoder
        self.conv1 = GCNConv(in_ch, hidden)
        self.conv2 = GCNConv(hidden, hidden)

    def encode(self, x, edge_index):
        x = F.relu(self.conv1(x, edge_index))
        x = F.dropout(x, 0.5, self.training)
        return self.conv2(x, edge_index)

    def decode(self, z, edge_index):
        # Dot-product scoring
        return (z[edge_index[0]] * z[edge_index[1]]).sum(dim=-1)

    def decode_all(self, z):
        # Score all pairs (for small graphs only)
        return torch.sigmoid((z @ z.t()))

model = LinkPredictor(dataset.num_features, 64)
optimizer = torch.optim.Adam(model.parameters(), lr=0.01)
```

## Training with negative sampling

```python
def train():
    model.train()
    optimizer.zero_grad()

    z = model.encode(train_data.x, train_data.edge_index)

    # Positive edges from the training split
    pos_edge_index = train_data.edge_label_index[:, train_data.edge_label == 1]

    # Sample equal number of negatives
    neg_edge_index = negative_sampling(
        edge_index=train_data.edge_index,
        num_nodes=train_data.num_nodes,
        num_neg_samples=pos_edge_index.size(1),
        method='sparse',
    )

    edge_index = torch.cat([pos_edge_index, neg_edge_index], dim=1)
    labels = torch.cat([
        torch.ones(pos_edge_index.size(1)),
        torch.zeros(neg_edge_index.size(1)),
    ])

    out = model.decode(z, edge_index)
    loss = F.binary_cross_entropy_with_logits(out, labels)
    loss.backward()
    optimizer.step()
    return float(loss)

for epoch in range(1, 101):
    loss = train()

print(f"Final training loss: {loss:.4f}")
```

## Evaluation: AUC-ROC and Average Precision

```python
from sklearn.metrics import roc_auc_score, average_precision_score

@torch.no_grad()
def evaluate(split_data):
    model.eval()
    z = model.encode(split_data.x, split_data.edge_index)
    out = model.decode(z, split_data.edge_label_index).sigmoid()
    return (
        roc_auc_score(split_data.edge_label.numpy(), out.numpy()),
        average_precision_score(split_data.edge_label.numpy(), out.numpy()),
```

## Manual negative sampling from scratch

Understanding what `negative_sampling` does under the hood:

```python
def my_negative_sampling(edge_index, num_nodes, num_samples):
    """Sample random non-edges."""
    # Build a set of existing edges for fast lookup
    existing = set(zip(edge_index[0].tolist(), edge_index[1].tolist()))
    neg_edges = []
    attempts = 0
    while len(neg_edges) < num_samples and attempts < num_samples * 10:
        u = torch.randint(0, num_nodes, (1,)).item()
        v = torch.randint(0, num_nodes, (1,)).item()
        if u != v and (u, v) not in existing and (v, u) not in existing:
            neg_edges.append([u, v])
        attempts += 1
    neg_edges = torch.tensor(neg_edges, dtype=torch.long).t()
    return neg_edges

neg = my_negative_sampling(data.edge_index, data.num_nodes, 100)
print(f"Sampled {neg.shape[1]} negative edges")
# Verify none are real edges
real_set = set(zip(data.edge_index[0].tolist(), data.edge_index[1].tolist()))
overlap = sum(1 for u,v in zip(neg[0].tolist(), neg[1].tolist()) if (u,v) in real_set)
print(f"Overlap with real edges: {overlap} (should be 0)")
```

## Try it Yourself

```python
import torch, torch.nn.functional as F
from torch_geometric.datasets import Planetoid
from torch_geometric.transforms import RandomLinkSplit
from torch_geometric.nn import SAGEConv
from torch_geometric.utils import negative_sampling
from sklearn.metrics import roc_auc_score

dataset = Planetoid(root='/tmp/Cora', name='Cora')
data = dataset[0]
transform = RandomLinkSplit(num_val=0.05, num_test=0.10, is_undirected=True,
                             add_negative_train_samples=False)
tr, va, te = transform(data)

import torch.nn as nn
class SAGE_LP(nn.Module):
    def __init__(self):
        super().__init__()
        self.c1 = SAGEConv(dataset.num_features, 64)
        self.c2 = SAGEConv(64, 64)
    def encode(self, x, ei):
        return self.c2(F.relu(self.c1(x, ei)), ei)
    def decode(self, z, ei):
        return (z[ei[0]] * z[ei[1]]).sum(-1)

model = SAGE_LP(); opt = torch.optim.Adam(model.parameters(), lr=0.005)
for _ in range(100):
    model.train(); opt.zero_grad()
    z = model.encode(tr.x, tr.edge_index)
    pos = tr.edge_label_index[:, tr.edge_label==1]
    neg = negative_sampling(tr.edge_index, tr.num_nodes, pos.size(1))
    ei = torch.cat([pos, neg], 1)
    lbl = torch.cat([torch.ones(pos.size(1)), torch.zeros(neg.size(1))])
    F.binary_cross_entropy_with_logits(model.decode(z,ei), lbl).backward(); opt.step()

model.eval()
with torch.no_grad():
    z = model.encode(te.x, te.edge_index)
    out = model.decode(z, te.edge_label_index).sigmoid()
print(f"Test AUC: {roc_auc_score(te.edge_label.numpy(), out.numpy()):.4f}")
```

## Exercise

Implement **corrupted-head negative sampling**: instead of random node pairs, for each
positive edge $(u,v)$, generate negatives by replacing $u$ with a random node (keeping $v$
fixed). Then also generate negatives by replacing $v$ (keeping $u$ fixed). Does this harder
negative strategy improve test AUC compared to uniform random negatives?

```python
def corrupt_head_tail(pos_edge_index, num_nodes):
    """For each positive (u,v), generate (rand,v) and (u,rand)."""
    E = pos_edge_index.size(1)
    rand_nodes = torch.randint(0, num_nodes, (E * 2,))
    # Corrupt head
    neg_head = torch.stack([rand_nodes[:E], pos_edge_index[1]])
    # Corrupt tail
    neg_tail = torch.stack([pos_edge_index[0], rand_nodes[E:]])
    return torch.cat([neg_head, neg_tail], dim=1)

# Use in training loop:
# neg_edge_index = corrupt_head_tail(pos_edge_index, train_data.num_nodes)
# This samples 2× more negatives but they are "harder" (share one endpoint with a positive).
# Typically improves test AUC by 0.5–1% on citation networks.
```
