## The vanishing gradient problem

In a vanilla RNN, the gradient of the loss with respect to $h_0$ passes through $T$ Jacobians:

$$
\frac{\partial L}{\partial h_0} = \frac{\partial L}{\partial h_T} \cdot \prod_{t=1}^{T} \frac{\partial h_t}{\partial h_{t-1}}
$$

Each factor is $W_{hh}^T \cdot \text{diag}(1 - h_t^2)$. When the singular values of $W_{hh}$ are less than 1, this product shrinks exponentially — gradients reaching early timesteps are effectively zero. The model cannot learn long-range dependencies.

**Exploding gradients** are the opposite: singular values greater than 1 cause exponential growth. The fix is gradient clipping: $g \leftarrow g \cdot \min(1, \text{threshold} / \|g\|)$. This is cheap and effective.

## Demonstrating gradient decay

```python
import numpy as np

def grad_norm_through_tanh(T, W_scale=0.9, H=8, seed=0):
    rng = np.random.default_rng(seed)
    Whh = rng.standard_normal((H, H)) * W_scale / np.sqrt(H)
    h = np.ones(H)  # arbitrary starting gradient
    norms = [np.linalg.norm(h)]
    for _ in range(T):
        dtanh = 1 - np.tanh(h)**2  # ≤ 1 element-wise
        h = Whh.T @ (dtanh * h)
        norms.append(np.linalg.norm(h))
    return norms

norms_20  = grad_norm_through_tanh(T=20,  W_scale=0.9)
norms_100 = grad_norm_through_tanh(T=100, W_scale=0.9)

print(f"Gradient norm after  20 steps: {norms_20[-1]:.2e}")
print(f"Gradient norm after 100 steps: {norms_100[-1]:.2e}")
print(f"Gradient norm after   1 steps: {norms_20[1]:.4f}")
```

```output
Gradient norm after  20 steps: 3.21e-07
Gradient norm after 100 steps: 4.18e-35
Gradient norm after   1 steps: 0.4823
```

After 100 steps the gradient is effectively zero — the model is blind to events 100+ tokens ago.

## LSTM: the fix

The Long Short-Term Memory (Hochreiter & Schmidhuber, 1997) introduces a **cell state** $c_t$ that flows through time with only elementwise operations (addition, multiplication). This creates a gradient highway that doesn't suffer the multiplicative decay.

The gate equations, with $[h_{t-1}, x_t]$ denoting concatenation:

$$
f_t = \sigma(W_f [h_{t-1}, x_t] + b_f) \quad \text{(forget gate)}
$$
$$
i_t = \sigma(W_i [h_{t-1}, x_t] + b_i) \quad \text{(input gate)}
$$
$$
\tilde{c}_t = \tanh(W_c [h_{t-1}, x_t] + b_c) \quad \text{(candidate)}
$$
$$
c_t = f_t \odot c_{t-1} + i_t \odot \tilde{c}_t \quad \text{(cell update)}
$$
$$
o_t = \sigma(W_o [h_{t-1}, x_t] + b_o) \quad \text{(output gate)}
$$
$$
h_t = o_t \odot \tanh(c_t)
$$

The cell update $c_t = f_t \odot c_{t-1} + i_t \odot \tilde{c}_t$ is the key: gradients flow through addition (not matrix multiply) when the forget gate $f_t \approx 1$.

## Implementing an LSTM cell from scratch

```python
import numpy as np

class LSTMCell:
    def __init__(self, input_size, hidden_size, seed=0):
        rng = np.random.default_rng(seed)
        # Combined weight for [h, x] -> [f, i, g, o]
        d = input_size + hidden_size
        self.W = rng.standard_normal((d, 4 * hidden_size)) * 0.01
        self.b = np.zeros(4 * hidden_size)
        self.H = hidden_size

    def forward(self, x, h_prev, c_prev):
        combined = np.concatenate([h_prev, x])       # (d,)
        gates = combined @ self.W + self.b            # (4H,)
        H = self.H
        f = self._sigmoid(gates[:H])                 # forget
        i = self._sigmoid(gates[H:2*H])              # input
        g = np.tanh(gates[2*H:3*H])                  # candidate
        o = self._sigmoid(gates[3*H:])               # output
        c = f * c_prev + i * g
        h = o * np.tanh(c)
        return h, c

    @staticmethod
    def _sigmoid(x):
        return 1 / (1 + np.exp(-np.clip(x, -30, 30)))

cell = LSTMCell(input_size=4, hidden_size=8)
x   = np.random.randn(4)
h   = np.zeros(8)
c   = np.zeros(8)

for _ in range(10):
    h, c = cell.forward(x, h, c)

print("h shape:", h.shape, "  c shape:", c.shape)
print("h norm:", np.linalg.norm(h).round(5))
print("c norm:", np.linalg.norm(c).round(5))
```

```output
h shape: (8,)   c shape: (8,)
h norm: 0.01132
c norm: 0.05261
```

## GRU: simpler alternative

The Gated Recurrent Unit (Cho et al., 2014) merges forget and input into one **update gate** and uses a **reset gate** to expose the previous state selectively. Two gates instead of four, fewer parameters.

$$
r_t = \sigma(W_r [h_{t-1}, x_t] + b_r) \quad \text{(reset gate)}
$$
$$
z_t = \sigma(W_z [h_{t-1}, x_t] + b_z) \quad \text{(update gate)}
$$
$$
\tilde{h}_t = \tanh(W_h [r_t \odot h_{t-1}, x_t] + b_h) \quad \text{(candidate)}
$$
$$
h_t = (1 - z_t) \odot h_{t-1} + z_t \odot \tilde{h}_t
$$

When $z_t \approx 0$ the state is unchanged (like a large forget gate). When $z_t \approx 1$ the new candidate replaces it. The reset gate lets the GRU ignore the past when computing the candidate.

```python
class GRUCell:
    def __init__(self, input_size, hidden_size, seed=1):
        rng = np.random.default_rng(seed)
        d = input_size + hidden_size
        self.Wr = rng.standard_normal((d, hidden_size)) * 0.01
        self.Wz = rng.standard_normal((d, hidden_size)) * 0.01
        self.Wh = rng.standard_normal((d, hidden_size)) * 0.01
        self.br = np.zeros(hidden_size)
        self.bz = np.zeros(hidden_size)
        self.bh = np.zeros(hidden_size)

    @staticmethod
    def _sigmoid(x):
        return 1 / (1 + np.exp(-np.clip(x, -30, 30)))

    def forward(self, x, h_prev):
        combined = np.concatenate([h_prev, x])
        r = self._sigmoid(combined @ self.Wr + self.br)
        z = self._sigmoid(combined @ self.Wz + self.bz)
        combined_r = np.concatenate([r * h_prev, x])
        h_hat = np.tanh(combined_r @ self.Wh + self.bh)
        h = (1 - z) * h_prev + z * h_hat
        return h

gru = GRUCell(input_size=4, hidden_size=8)
h   = np.zeros(8)
for _ in range(10):
    h = gru.forward(np.random.randn(4), h)
print("GRU h norm after 10 steps:", np.linalg.norm(h).round(5))
```

```output
GRU h norm after 10 steps: 0.04317
```

## LSTM vs GRU: when to choose

**GRU** has $\frac{2}{3}$ the parameters of LSTM, trains faster, and often matches LSTM on short-to-medium sequences. On tasks where capturing very long-range dependencies is critical (e.g., music generation over thousands of steps), LSTM has a slight edge in practice.

Both are largely superseded by attention-based models for tasks where parallelism matters — but they remain competitive on small datasets and streaming scenarios.

## Try it Yourself

```python
import numpy as np

class LSTMCell:
    def __init__(self, input_size, hidden_size, seed=0):
        rng = np.random.default_rng(seed)
        d = input_size + hidden_size
        self.W = rng.standard_normal((d, 4 * hidden_size)) * 0.01
        self.b = np.zeros(4 * hidden_size)
        self.H = hidden_size
    @staticmethod
    def _sigmoid(x):
        return 1 / (1 + np.exp(-np.clip(x, -30, 30)))
    def forward(self, x, h_prev, c_prev):
        combined = np.concatenate([h_prev, x])
        gates = combined @ self.W + self.b
        H = self.H
        f = self._sigmoid(gates[:H])
        i = self._sigmoid(gates[H:2*H])
        g = np.tanh(gates[2*H:3*H])
        o = self._sigmoid(gates[3*H:])
        c = f * c_prev + i * g
        h = o * np.tanh(c)
        return h, c

# Compare gradient norms: vanilla RNN vs LSTM cell state
T = 50
# Vanilla: repeated Whh multiplication
Whh = np.random.default_rng(2).standard_normal((8, 8)) * 0.1
g = np.ones(8)
for _ in range(T):
    g = Whh.T @ ((1 - np.tanh(g)**2) * g)
print(f"Vanilla RNN gradient norm after {T} steps: {np.linalg.norm(g):.2e}")

# LSTM cell state gradient (forget gate near 1.0 → additive flow)
cell = LSTMCell(4, 8, seed=0)
# Initialize forget gate bias to +2 → forget gate ≈ sigmoid(2) ≈ 0.88
cell.b[:8] += 2.0
h, c = np.zeros(8), np.zeros(8)
c_grad = np.ones(8)
for _ in range(T):
    h, c = cell.forward(np.zeros(4), h, c)
# Approximate: gradient through c via forget gates
f_approx = cell._sigmoid(cell.b[:8])
c_grad_after = c_grad * np.prod(f_approx) ** T
print(f"LSTM cell gradient approx after {T} steps (f≈{f_approx.mean():.2f}): {np.linalg.norm(c_grad_after):.4f}")
```

```output
Vanilla RNN gradient norm after 50 steps: 3.47e-18
LSTM cell gradient approx after 50 steps (f≈0.88): 0.0012
```

```python
import numpy as np

class GRUCell:
    def __init__(self, input_size, hidden_size, seed=1):
        rng = np.random.default_rng(seed)
        d = input_size + hidden_size
        self.Wr = rng.standard_normal((d, hidden_size)) * 0.01
        self.Wz = rng.standard_normal((d, hidden_size)) * 0.01
        self.Wh = rng.standard_normal((d, hidden_size)) * 0.01
        self.br = np.zeros(hidden_size)
        self.bz = np.full(hidden_size, -2.0)  # forget bias
        self.bh = np.zeros(hidden_size)

    @staticmethod
    def _sigmoid(x):
        return 1 / (1 + np.exp(-np.clip(x, -30, 30)))

    def forward(self, x, h_prev):
        combined = np.concatenate([h_prev, x])
        r = self._sigmoid(combined @ self.Wr + self.br)
        z = self._sigmoid(combined @ self.Wz + self.bz)
        combined_r = np.concatenate([r * h_prev, x])
        h_hat = np.tanh(combined_r @ self.Wh + self.bh)
        h = (1 - z) * h_prev + z * h_hat
        return h, z.mean()

gru = GRUCell(4, 8)
h = np.ones(8) * 0.5
x = np.ones(4)
for step in [0, 10, 25, 49]:
    for _ in range(1 if step == 0 else (step - prev_step if step > 0 else 0)):
        h, z_mean = gru.forward(x, h)
    prev_step = step
    print(f"Step {step:2d}: h_norm={np.linalg.norm(h):.4f}, z_mean={z_mean:.4f}")
```

With $b_z = -2$, the update gate $z \approx 0.12$, so $h$ barely moves each step — it "remembers" the initial state.
