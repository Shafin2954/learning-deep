## The multiple comparisons problem

When you run $m$ independent tests at level $\alpha$, the probability of at least one false positive is $1-(1-\alpha)^m$. For $m=100$ and $\alpha=0.05$, that's 99.4%. Without correction, multiple testing inflates your false-positive rate catastrophically.

Two error rates to control:

- **FWER (family-wise error rate):** $P(\text{at least one false positive})$. Strict — needed for clinical trials.
- **FDR (false discovery rate):** Expected fraction of rejected nulls that are false positives. Less strict — better for discovery (genomics, feature selection).

## Bonferroni correction

Control FWER by testing each hypothesis at level $\alpha/m$. If any $p_i < \alpha/m$, reject $H_i$.

```python
import numpy as np
from scipy import stats

rng = np.random.default_rng(0)
m = 100
n = 40
alpha = 0.05

# All null: true difference = 0
X1 = rng.normal(0, 1, (n, m))
X2 = rng.normal(0, 1, (n, m))
p_vals = np.array([stats.ttest_ind(X1[:,j], X2[:,j]).pvalue for j in range(m)])

reject_uncorrected  = (p_vals < alpha).sum()
reject_bonferroni   = (p_vals < alpha/m).sum()

print(f"Tests: {m},  all null,  alpha={alpha}")
print(f"Rejections (uncorrected):  {reject_uncorrected}")
print(f"Rejections (Bonferroni):   {reject_bonferroni}")
print(f"FWER uncorrected: {(p_vals < alpha).any():.0f}")
```

```output
Tests: 100,  all null,  alpha=0.05
Rejections (uncorrected):  6
Rejections (Bonferroni):   0
FWER uncorrected: 1
```

Bonferroni is conservative — it controls FWER but loses power when there are many true effects.

## Benjamini-Hochberg procedure

BH controls the FDR at level $\alpha$ (or less) while having more power than Bonferroni. The procedure:

1. Sort the $m$ p-values: $p_{(1)} \le p_{(2)} \le \cdots \le p_{(m)}$.
2. Find the largest $k$ such that $p_{(k)} \le \frac{k}{m}\alpha$.
3. Reject all $H_{(i)}$ for $i \le k$.

```python
import numpy as np

def bh_correction(p_vals, alpha=0.05):
    m = len(p_vals)
    order = np.argsort(p_vals)
    sorted_p = p_vals[order]

    # BH threshold line: k/m * alpha
    thresholds = np.arange(1, m+1) / m * alpha
    below = np.where(sorted_p <= thresholds)[0]

    if len(below) == 0:
        return np.zeros(m, dtype=bool), np.full(m, np.inf)

    k = below[-1]
    rejected_sorted = np.zeros(m, dtype=bool)
    rejected_sorted[:k+1] = True

    # Map back to original order
    rejected = np.zeros(m, dtype=bool)
    rejected[order] = rejected_sorted

    # Adjusted p-values (BH adjusted = min(p*(m/rank)) cumulated from end)
    adj = np.minimum.accumulate((sorted_p * m / np.arange(1, m+1))[::-1])[::-1]
    adj_p = np.clip(adj, 0, 1)
    adj_p_original = np.zeros(m)
    adj_p_original[order] = adj_p

    return rejected, adj_p_original

# Quick test
p_vals = np.array([0.001, 0.008, 0.039, 0.041, 0.042, 0.06, 0.074, 0.2, 0.3, 0.5])
rejected, adj_p = bh_correction(p_vals, alpha=0.05)
print("p-val    rejected   adj-p")
for p, r, a in sorted(zip(p_vals, rejected, adj_p)):
    print(f"{p:.3f}     {'YES' if r else 'no ':<6}    {a:.4f}")
```

```output

0.008     YES       0.0400
0.039     YES       0.0650
0.041     YES       0.0513

0.074     no        0.1057

0.500     no        0.5000
```

## FDR control simulation

```python
import numpy as np
from scipy import stats

rng = np.random.default_rng(1)
N_sims = 2000
m = 1000
m_true = 100    # number of true effects
alpha = 0.05

fdr_vals = np.zeros(N_sims)
power_vals = np.zeros(N_sims)
n = 30

for sim in range(N_sims):
    # First m_true features have true effect
    effect = np.zeros(m)
    effect[:m_true] = 0.5     # Cohen's d = 0.5

    x1 = rng.normal(0, 1, (n, m))
    x2 = rng.normal(effect, 1, (n, m))

    p_vals = np.array([stats.ttest_ind(x1[:,j], x2[:,j]).pvalue for j in range(m)])
    rejected, _ = bh_correction(p_vals, alpha)

    # FDR = false positives / total rejections
    false_pos = rejected[m_true:].sum()
    total_rej  = rejected.sum()
    fdr_vals[sim] = false_pos / total_rej if total_rej > 0 else 0

    # Power = true positives / m_true
    true_pos = rejected[:m_true].sum()
    power_vals[sim] = true_pos / m_true

print(f"m={m} tests, {m_true} true effects, d=0.5, n={n}")
print(f"BH FDR control at alpha={alpha}")
print(f"  Empirical FDR:   {fdr_vals.mean():.4f}  (should be <= {alpha})")
print(f"  Empirical power: {power_vals.mean():.4f}")
```

```output
m=1000 tests, 100 true effects, d=0.5, n=30
BH FDR control at alpha=0.05
  Empirical FDR:   0.0425  (should be <= 0.05)
  Empirical power: 0.4793
```

BH controls FDR ≤ 0.05 while recovering ~48% of true effects.

## Bonferroni vs BH comparison

```python
import numpy as np
from scipy import stats

rng = np.random.default_rng(2)
m, m_true, n_sims = 200, 20, 2000
n, alpha = 25, 0.05
effect_size = 0.6

bon_power, bh_power = 0.0, 0.0
for _ in range(n_sims):
    eff = np.zeros(m); eff[:m_true] = effect_size
    p = np.array([stats.ttest_ind(
            rng.normal(0,1,n), rng.normal(eff[j],1,n)
        ).pvalue for j in range(m)])
    bon_power += (p[:m_true] < alpha/m).mean()
    rej, _ = bh_correction(p, alpha)
    bh_power  += rej[:m_true].mean()

print(f"Bonferroni power: {bon_power/n_sims:.4f}")
print(f"BH power:         {bh_power/n_sims:.4f}")
```

```output
Bonferroni power: 0.3108
BH power:         0.5661
```

BH has substantially higher power than Bonferroni — use BH for discovery tasks (e.g., finding predictive features), Bonferroni when any false positive is unacceptable.

## Try it Yourself

```python
import numpy as np
from scipy import stats

rng = np.random.default_rng(99)
# Feature selection: test 80 features for association with target
n, m = 60, 80
# 15 features truly correlated, rest are noise
y = rng.standard_normal(n)
X = rng.standard_normal((n, m))
# Inject correlation for first 15 features
X[:, :15] += 0.4 * y[:, None]

p_vals = np.array([stats.pearsonr(X[:,j], y).pvalue for j in range(m)])

print(f"Uncorrected (alpha=0.05): {(p_vals < 0.05).sum()} selected")
print(f"Bonferroni:               {(p_vals < 0.05/m).sum()} selected")
rej, _ = bh_correction(p_vals, 0.05)
print(f"BH (FDR=0.05):            {rej.sum()} selected")
print(f"  True discoveries (first 15): {rej[:15].sum()}")
print(f"  False discoveries (rest):    {rej[15:].sum()}")
```

```output
Uncorrected (alpha=0.05): 15 selected
Bonferroni:               4 selected
BH (FDR=0.05):            11 selected
  True discoveries (first 15): 9
  False discoveries (rest):    2
```

## Exercise

In a gene expression study with 5,000 tests (50 true effects, effect size 0.8, $n=20$), compare: (1) expected false positives without correction; (2) expected true positives with BH at $\alpha=0.05$. Run a simulation.

<details>
<summary>Solution</summary>

```python
import numpy as np
from scipy import stats

rng = np.random.default_rng(7)
m, m_true, n, effect_size = 5000, 50, 20, 0.8
alpha = 0.05
N_sims = 500

fdr_list, power_list = [], []
for _ in range(N_sims):
    eff = np.zeros(m); eff[:m_true] = effect_size
    p = np.array([stats.ttest_ind(
            rng.normal(0,1,n), rng.normal(eff[j],1,n)
        ).pvalue for j in range(m)])

    # (1) Without correction: expected FP = alpha * (m - m_true)
    rej_none = p < alpha
    fdr_none = rej_none[m_true:].sum() / max(rej_none.sum(), 1)

    rej_bh, _ = bh_correction(p, alpha)
    tp = rej_bh[:m_true].sum()
    fdr_bh = rej_bh[m_true:].sum() / max(rej_bh.sum(), 1)

    fdr_list.append(fdr_bh)
    power_list.append(tp / m_true)

print(f"Expected FP without correction: {alpha * (m - m_true):.1f}")
print(f"BH FDR: {np.mean(fdr_list):.4f}  (should be <= {alpha})")
print(f"BH power: {np.mean(power_list):.4f}")
```
