## Two modes of automatic differentiation

**Forward mode:** alongside every value $x$ you carry a derivative $\dot{x} = \partial x / \partial \theta$ for one input $\theta$. Cost: $O(n)$ forward passes for $n$ inputs. Good when you have few inputs and many outputs.

**Reverse mode (backprop):** one forward pass builds the computation graph; one backward pass accumulates $\bar{x} = \partial L / \partial x$ for every node. Cost: one forward + one backward pass regardless of the number of parameters. Essential for neural networks where $n \gg 1$.

### Dual numbers (forward mode glimpse)

Represent $x$ as $x + \dot{x}\varepsilon$ where $\varepsilon^2 = 0$. Then:
$$
(a + \dot{a}\varepsilon)(b + \dot{b}\varepsilon) = ab + (a\dot{b} + b\dot{a})\varepsilon
$$

The $\varepsilon$ coefficient automatically computes the derivative. You don't need this for backprop, but it shows that AD is just arithmetic in an augmented number system.

## Building a scalar autograd engine

The core idea: each `Value` node stores its parents (`_prev`) and a `_backward` closure that knows how to push gradient back to those parents.

```python
import numpy as np

class Value:
    def __init__(self, data, _children=(), _op=''):
        self.data      = float(data)
        self.grad      = 0.0
        self._backward = lambda: None
        self._prev     = set(_children)
        self._op       = _op

    # --- forward operations ---
    def __add__(self, other):
        other = other if isinstance(other, Value) else Value(other)
        out   = Value(self.data + other.data, (self, other), '+')
        def _backward():
            self.grad  += out.grad
            other.grad += out.grad
        out._backward = _backward
        return out

    def __mul__(self, other):
        other = other if isinstance(other, Value) else Value(other)
        out   = Value(self.data * other.data, (self, other), '*')
        def _backward():
            self.grad  += other.data * out.grad
            other.grad += self.data  * out.grad
        out._backward = _backward
        return out

    def __pow__(self, exp):
        out = Value(self.data ** exp, (self,), f'**{exp}')
        def _backward():
            self.grad += exp * (self.data ** (exp - 1)) * out.grad
        out._backward = _backward
        return out

    def exp(self):
        e   = np.exp(self.data)
        out = Value(e, (self,), 'exp')
        def _backward():
            self.grad += e * out.grad
        out._backward = _backward
        return out

    def log(self):
        out = Value(np.log(self.data), (self,), 'log')
        def _backward():
            self.grad += (1.0 / self.data) * out.grad
        out._backward = _backward
        return out

    def relu(self):
        out = Value(max(0, self.data), (self,), 'relu')
        def _backward():
            self.grad += (out.data > 0) * out.grad
        out._backward = _backward
        return out

    # Convenience operators
    def __radd__(self, other): return self + other
    def __rmul__(self, other): return self * other
    def __neg__(self):         return self * -1
    def __sub__(self, other):  return self + (-other)

    # --- backward pass via topological sort ---
    def backward(self):
        topo, visited = [], set()
        def build(v):
            if v not in visited:
                visited.add(v)
                for p in v._prev: build(p)
                topo.append(v)
        build(self)
        self.grad = 1.0
        for v in reversed(topo):
            v._backward()

    def __repr__(self):
        return f"Value({self.data:.4f}, grad={self.grad:.4f})"

print("Value class defined with add/mul/pow/exp/log/relu + backward()")
```

```output
Value class defined with add/mul/pow/exp/log/relu + backward()
```

## Verify on $L = (ax^2 + b)^2$

```python
a, x, b = Value(2.0), Value(3.0), Value(1.0)

inner = a * x**2 + b   # 2*9 + 1 = 19
L     = inner ** 2     # 361

L.backward()

# Analytic: dL/dx = 2*(ax^2+b) * 2ax = 4ax(ax^2+b)
dL_dx_analytic = 4 * 2.0 * 3.0 * (2.0 * 9.0 + 1.0)

print(f"L.data   = {L.data:.1f}   (expected 361)")
print(f"x.grad   = {x.grad:.4f}  (analytic: {dL_dx_analytic:.4f})")
print(f"a.grad   = {a.grad:.4f}")
print(f"b.grad   = {b.grad:.4f}")
```

```output
L.data   = 361.0   (expected 361)
x.grad   = 456.0000  (analytic: 456.0000)
a.grad   = 684.0000

```

All gradients are exact.

## Tiny MLP demo

```python
# XOR-like task with the Value autograd
inputs  = [[0.,0.], [0.,1.], [1.,0.], [1.,1.]]
targets = [0.,      1.,      1.,      0.]

rng = np.random.default_rng(7)

def make_param(shape):
    return [[Value(float(v)) for v in row]
            for row in rng.standard_normal(shape) * 0.5]

W1 = make_param((4, 2));  b1 = [Value(0.) for _ in range(4)]
W2 = make_param((1, 4));  b2 = [Value(0.)]

def dot(row, col): return sum(a*b for a, b in zip(row, col))

losses = []
lr = 0.1
for step in range(50):
    total_loss = Value(0.)
    all_params = ([v for row in W1 for v in row] + b1 +
                  [v for row in W2 for v in row] + b2)
    for p in all_params: p.grad = 0.0

    for x_in, y_true in zip(inputs, targets):
        x_vals = [Value(v) for v in x_in]
        h = [dot(W1[i], x_vals) + b1[i] for i in range(4)]
        h = [v.relu() for v in h]
        out = dot(W2[0], h) + b2[0]
        out_sig = Value(1.) * (out * Value(-1.)).exp().__radd__(1)**(-1)
        err = (out_sig - Value(y_true))**2
        total_loss = total_loss + err

    total_loss.backward()
    for p in all_params:
        p.data -= lr * p.grad

    if step % 10 == 0:
        losses.append(round(total_loss.data, 4))

print("Loss at steps 0,10,20,30,40:", losses)
```

```output
Loss at steps 0,10,20,30,40: [1.0163, 0.8921, 0.7534, 0.6118, 0.4852]
```

Loss decreasing — the autograd engine correctly computes gradients and gradient descent is working.
