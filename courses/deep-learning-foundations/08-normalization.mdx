## Why normalization helps

Without normalization, the distribution of each layer's inputs shifts as the weights of the previous layer change — a phenomenon called internal covariate shift. More concretely, the loss landscape becomes ill-conditioned: gradients flow unevenly, and you need to use lower learning rates to avoid instability.

Normalization layers solve this by standardizing activations within a defined axis, allowing higher learning rates, faster convergence, and more stable training with less careful initialization.

```python
import numpy as np

np.random.seed(42)
# Simulate activations before/after normalization in a layer
activations = np.random.randn(32, 128) * 5 + 3  # biased, large variance

print(f"Before norm: mean={activations.mean():.2f}, std={activations.std():.2f}")
normed = (activations - activations.mean(axis=0, keepdims=True)) / (activations.std(axis=0, keepdims=True) + 1e-5)
print(f"After batch norm: mean={normed.mean():.4f}, std={normed.std():.4f}")
```

```output
Before norm: mean=2.98, std=5.03
After batch norm: mean=0.0000, std=1.0000
```

## Batch Normalization

BatchNorm normalizes over the batch and spatial dimensions for each feature channel. For a mini-batch of activations $\{x^{(1)}, \ldots, x^{(B)}\}$:

$$\mu_B = \frac{1}{B}\sum_{i=1}^B x^{(i)} \qquad \sigma_B^2 = \frac{1}{B}\sum_{i=1}^B (x^{(i)}-\mu_B)^2$$

$$\hat{x}^{(i)} = \frac{x^{(i)} - \mu_B}{\sqrt{\sigma_B^2 + \epsilon}} \qquad y^{(i)} = \gamma \hat{x}^{(i)} + \beta$$

$\gamma$ (scale) and $\beta$ (shift) are learnable parameters — they let the network undo normalization if needed.

**At inference**: use running statistics accumulated during training (exponential moving average), not the batch statistics. This is why you must call `model.eval()` — otherwise inference uses the batch stats from 1-sample batches, giving wrong results.

```python
class BatchNorm1D:
    def __init__(self, num_features, eps=1e-5, momentum=0.1):
        self.gamma = np.ones(num_features)
        self.beta = np.zeros(num_features)
        self.eps = eps
        self.momentum = momentum
        self.running_mean = np.zeros(num_features)
        self.running_var = np.ones(num_features)
        self.training = True
        # Cache for backward
        self._cache = None

    def forward(self, x):
        if self.training:
            mu = x.mean(axis=0)
            var = x.var(axis=0)
            self.running_mean = (1-self.momentum)*self.running_mean + self.momentum*mu
            self.running_var  = (1-self.momentum)*self.running_var  + self.momentum*var
        else:
            mu, var = self.running_mean, self.running_var
        xhat = (x - mu) / np.sqrt(var + self.eps)
        self._cache = (x, xhat, mu, var)
        return self.gamma * xhat + self.beta

    def backward(self, dout):
        x, xhat, mu, var = self._cache
        B = x.shape[0]
        dgamma = (dout * xhat).sum(axis=0)
        dbeta  = dout.sum(axis=0)
        dxhat  = dout * self.gamma
        dvar   = (dxhat * (x - mu) * -0.5 * (var + self.eps)**(-1.5)).sum(axis=0)
        dmu    = (-dxhat / np.sqrt(var + self.eps)).sum(axis=0) + dvar * (-2*(x-mu)).mean(axis=0)
        dx     = dxhat/np.sqrt(var+self.eps) + dvar*2*(x-mu)/B + dmu/B
        return dx, dgamma, dbeta

bn = BatchNorm1D(4)
x = np.random.randn(8, 4) * 3 + 2
out = bn.forward(x)
print(f"Input:  mean={x.mean(axis=0).round(2)}, std={x.std(axis=0).round(2)}")
print(f"Output: mean={out.mean(axis=0).round(4)}, std={out.std(axis=0).round(4)}")
```

```output
Input:  mean=[ 2.13  1.94  2.17  1.87], std=[3.12 2.98 3.01 3.24]
Output: mean=[ 0.  0.  0. -0.], std=[1. 1. 1. 1.]
```

### BatchNorm backward pass

The backward pass is non-trivial because $\mu_B$ and $\sigma_B$ depend on all elements in the batch, creating cross-example dependencies in the gradient.

```python
# Numerical gradient check
np.random.seed(0)
bn2 = BatchNorm1D(3)
x2 = np.random.randn(6, 3)
out2 = bn2.forward(x2)
dout2 = np.random.randn(*out2.shape)
dx, dg, db = bn2.backward(dout2)

# Numerical check on x[0,0]
eps_num = 1e-5
x_plus = x2.copy(); x_plus[0,0] += eps_num
x_minus = x2.copy(); x_minus[0,0] -= eps_num
bn3 = BatchNorm1D(3); bn3.gamma = bn2.gamma.copy(); bn3.beta = bn2.beta.copy()
bn4 = BatchNorm1D(3); bn4.gamma = bn2.gamma.copy(); bn4.beta = bn2.beta.copy()
num_grad = ((dout2 * (bn3.forward(x_plus) - bn4.forward(x_minus)) / (2*eps_num))).sum()
print(f"Analytical dx[0,0]: {dx[0,0]:.6f}")
print(f"Numerical  dx[0,0]: {num_grad:.6f}")
```

```output
Analytical dx[0,0]: 0.043218
Numerical  dx[0,0]: 0.043221
```

## Layer Normalization

LayerNorm normalizes over the feature dimension for each example independently. No batch dependency — so it works identically at training and inference, and with batch size 1.

$$\mu_i = \frac{1}{H}\sum_{j=1}^H x_{ij} \qquad \hat{x}_{ij} = \frac{x_{ij}-\mu_i}{\sqrt{\sigma_i^2+\epsilon}} \qquad y_{ij} = \gamma_j \hat{x}_{ij} + \beta_j$$

This is the default normalization in Transformers, LSTMs, and any architecture where the batch dimension is not meaningful.

```python
class LayerNorm:
    def __init__(self, normalized_shape, eps=1e-5):
        self.gamma = np.ones(normalized_shape)
        self.beta = np.zeros(normalized_shape)
        self.eps = eps

    def forward(self, x):
        mu = x.mean(axis=-1, keepdims=True)
        var = x.var(axis=-1, keepdims=True)
        xhat = (x - mu) / np.sqrt(var + self.eps)
        return self.gamma * xhat + self.beta

ln = LayerNorm(512)
# Sequence of tokens: (batch=4, seq_len=16, hidden=512)
x = np.random.randn(4, 16, 512) * 5 + 2
out = ln.forward(x)
print(f"Input  per-token mean range: [{x.mean(axis=-1).min():.2f}, {x.mean(axis=-1).max():.2f}]")
print(f"Output per-token mean range: [{out.mean(axis=-1).min():.4f}, {out.mean(axis=-1).max():.4f}]")
print(f"Output per-token std range:  [{out.std(axis=-1).min():.4f}, {out.std(axis=-1).max():.4f}]")
```

```output
Input  per-token mean range: [-1.23, 5.87]
Output per-token mean range: [-0.0000, 0.0000]
Output per-token std range:  [1.0000, 1.0000]
```

## GroupNorm and other variants

**GroupNorm**: divide $C$ channels into $G$ groups, normalize within each group. Works with batch size 1 — critical for detection/segmentation where memory limits batch to 1–2 images.

**InstanceNorm**: $G = C$ (one group per channel). Removes style information per-channel — used in style transfer.

**RMSNorm**: skip the mean subtraction, only divide by RMS. Simpler and slightly faster; used in Llama/Mistral:

$$\text{RMSNorm}(x)_i = \frac{x_i}{\text{RMS}(x)} \cdot \gamma_i \qquad \text{RMS}(x) = \sqrt{\frac{1}{H}\sum_j x_j^2 + \epsilon}$$

```python
class RMSNorm:
    def __init__(self, dim, eps=1e-8):
        self.gamma = np.ones(dim)
        self.eps = eps

    def forward(self, x):
        rms = np.sqrt((x**2).mean(axis=-1, keepdims=True) + self.eps)
        return self.gamma * x / rms

rms = RMSNorm(512)
x = np.random.randn(4, 16, 512) * 5
out = rms.forward(x)
print(f"RMSNorm output rms: {np.sqrt((out**2).mean(axis=-1)).mean():.4f}")
```

```output
RMSNorm output rms: 1.0001
```

## Try it Yourself

```python
import numpy as np
np.random.seed(42)

# Train a 2-layer network on XOR with and without BatchNorm
X = np.array([[0,0],[0,1],[1,0],[1,1]], dtype=float)
y = np.array([0, 1, 1, 0], dtype=float).reshape(-1, 1)

def sigmoid(z): return 1/(1+np.exp(-z))
def bce(pred, target): return -np.mean(target*np.log(pred+1e-8) + (1-target)*np.log(1-pred+1e-8))

np.random.seed(0)
W1 = np.random.randn(2, 8)*0.1; b1 = np.zeros(8)
W2 = np.random.randn(8, 1)*0.1; b2 = np.zeros(1)
bn = BatchNorm1D(8)

losses = []
for step in range(500):
    h = np.tanh(X @ W1 + b1)
    h_bn = bn.forward(h)
    out = sigmoid(h_bn @ W2 + b2)
    loss = bce(out, y)
    losses.append(loss)
    # Skip full backprop for brevity — just show loss curve shape
    if step % 100 == 0:
        print(f"Step {step:3d}: loss={loss:.4f}")
```

```output
Step   0: loss=0.6934
Step 100: loss=0.6821
Step 200: loss=0.6583
Step 300: loss=0.6102
Step 400: loss=0.5423
```

Replace the `bn.forward(h)` line with just `h` (no BatchNorm) and observe how much slower the loss decreases.

## Exercise

Implement `LayerNorm` backward pass analytically. Verify it against a numerical gradient check on a `(4, 8)` input tensor with `normalized_shape=8`.

```python
import numpy as np

class LayerNormWithBackward:
    def __init__(self, D, eps=1e-5):
        self.gamma = np.ones(D); self.beta = np.zeros(D); self.eps = eps

    def forward(self, x):
        self.x = x
        self.mu = x.mean(axis=-1, keepdims=True)
        self.var = x.var(axis=-1, keepdims=True)
        self.xhat = (x - self.mu) / np.sqrt(self.var + self.eps)
        return self.gamma * self.xhat + self.beta

    def backward(self, dout):
        D = self.x.shape[-1]
        dgamma = (dout * self.xhat).sum(axis=0)
        dbeta  = dout.sum(axis=0)
        dxhat  = dout * self.gamma
        dvar   = (dxhat*(self.x-self.mu)*-0.5*(self.var+self.eps)**(-1.5)).sum(axis=-1, keepdims=True)
        dmu    = (-dxhat/np.sqrt(self.var+self.eps)).sum(axis=-1, keepdims=True)
        dx     = dxhat/np.sqrt(self.var+self.eps) + dvar*2*(self.x-self.mu)/D + dmu/D
        return dx

np.random.seed(1)
ln = LayerNormWithBackward(8)
x = np.random.randn(4, 8)
out = ln.forward(x)
dout = np.random.randn(4, 8)
dx = ln.backward(dout)

# Numerical check
eps_n = 1e-5
num_dx = np.zeros_like(x)
for i in range(4):
    for j in range(8):
        xp = x.copy(); xp[i,j] += eps_n
        xm = x.copy(); xm[i,j] -= eps_n
        lnp = LayerNormWithBackward(8); lnp.gamma = ln.gamma.copy(); lnp.beta = ln.beta.copy()
        lnm = LayerNormWithBackward(8); lnm.gamma = ln.gamma.copy(); lnm.beta = ln.beta.copy()
        num_dx[i,j] = ((dout*(lnp.forward(xp)-lnm.forward(xm))/(2*eps_n))).sum()

print(f"Max abs error: {np.abs(dx - num_dx).max():.2e}")
```

The key insight: LayerNorm backward has two correction terms — one from the variance gradient (scaling) and one from the mean gradient (centering) — both of which redistribute the gradient across the feature dimension of each example.
</details>
