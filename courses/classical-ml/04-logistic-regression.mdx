## From regression to classification

Linear regression predicts a real value. For binary classification, you need a probability in $(0,1)$. Apply the **sigmoid** function to the linear output:

$$
\sigma(z) = \frac{1}{1 + e^{-z}}, \quad z = w^Tx + b
$$

$\sigma$ maps any real number to $(0,1)$: $\sigma(0)=0.5$, $\sigma(\infty)=1$, $\sigma(-\infty)=0$. The predicted probability that $y=1$ is $\hat{p} = \sigma(w^Tx)$.

```python
import numpy as np
import matplotlib.pyplot as plt

z = np.linspace(-6, 6, 200)
sigma = 1 / (1 + np.exp(-z))

fig, ax = plt.subplots(figsize=(6, 3))
ax.plot(z, sigma, color='steelblue', linewidth=2)
ax.axhline(0.5, color='gray', linestyle='--', linewidth=0.8)
ax.axvline(0, color='gray', linestyle='--', linewidth=0.8)
ax.set_xlabel('z')
ax.set_ylabel('σ(z)')
ax.set_title('Sigmoid function')
plt.tight_layout()
plt.show()
```

```output
[Figure: sigmoid S-curve from 0 to 1 over z in [-6, 6], crossing 0.5 at z=0]
```

## Binary cross-entropy loss

Using MSE $\|y - \hat{p}\|^2$ with sigmoid leads to a non-convex loss. The correct loss is **binary cross-entropy**, which comes from maximum likelihood estimation of a Bernoulli distribution:

$$
L(w) = -\frac{1}{n}\sum_{i=1}^n \left[y_i \log \hat{p}_i + (1 - y_i)\log(1 - \hat{p}_i)\right]
$$

When $y_i=1$ and $\hat{p}_i\to 0$, the loss $\to\infty$ — a heavy penalty for confident wrong predictions. When $\hat{p}_i\to 1$, the loss $\to 0$.

### Gradient derivation

Let $z_i = w^Tx_i$, $\hat{p}_i = \sigma(z_i)$. Differentiating $L$ with respect to $w$:

$$
\frac{\partial L}{\partial w} = \frac{1}{n}\sum_{i=1}^n (\hat{p}_i - y_i)x_i = \frac{1}{n}X^T(\hat{p} - y)
$$

This is the same form as linear regression — the error times the features — but $\hat{p}=\sigma(Xw)$ instead of $Xw$.

## Implementing from scratch

```python
import numpy as np
from sklearn.datasets import make_classification
from sklearn.preprocessing import StandardScaler
from sklearn.model_selection import train_test_split

X, y = make_classification(n_samples=300, n_features=2, n_redundant=0,
                             n_clusters_per_class=1, random_state=42)
X = StandardScaler().fit_transform(X)
X_tr, X_te, y_tr, y_te = train_test_split(X, y, test_size=0.2, random_state=0)

def sigmoid(z):
    return 1 / (1 + np.exp(-z))

def logistic_fit(X, y, lr=0.1, epochs=500):
    n, p = X.shape
    w = np.zeros(p)
    b = 0.0
    losses = []
    for _ in range(epochs):
        p_hat = sigmoid(X @ w + b)
        loss = -np.mean(y * np.log(p_hat + 1e-15) + (1 - y) * np.log(1 - p_hat + 1e-15))
        losses.append(loss)
        dw = (X.T @ (p_hat - y)) / n
        db = np.mean(p_hat - y)
        w -= lr * dw
        b -= lr * db
    return w, b, losses

w, b, losses = logistic_fit(X_tr, y_tr)
preds = (sigmoid(X_te @ w + b) >= 0.5).astype(int)
acc = np.mean(preds == y_te)
print(f"Scratch accuracy: {acc:.3f}, final loss: {losses[-1]:.4f}")
```

```output
Scratch accuracy: 0.933, final loss: 0.1782
```

## Softmax for multiclass

For $K$ classes, replace sigmoid with softmax:

$$
p_k = \frac{e^{z_k}}{\sum_{j=1}^K e^{z_j}}, \quad z_k = w_k^T x + b_k
$$

The loss is categorical cross-entropy: $L = -\frac{1}{n}\sum_i \log p_{y_i}$. Logistic regression is the special case $K=2$.

```python
from sklearn.linear_model import LogisticRegression
from sklearn.datasets import load_iris

X_iris, y_iris = load_iris(return_X_y=True)
X_iris = StandardScaler().fit_transform(X_iris)
X_tr2, X_te2, y_tr2, y_te2 = train_test_split(X_iris, y_iris, test_size=0.2, random_state=0)

sk3 = LogisticRegression(multi_class='multinomial', max_iter=500).fit(X_tr2, y_tr2)
print(f"3-class accuracy: {sk3.score(X_te2, y_te2):.3f}")
print(f"Classes: {sk3.classes_}, coef shape: {sk3.coef_.shape}")
```

```output
3-class accuracy: 0.967
Classes: [0 1 2], coef shape: (3, 4)
```

## Try it Yourself

```python
import numpy as np

def sigmoid(z):
    return 1 / (1 + np.exp(-z))

# 1D logistic regression — plot the decision boundary
rng = np.random.default_rng(5)
X1 = np.concatenate([rng.normal(-1, 0.5, 50), rng.normal(1, 0.5, 50)]).reshape(-1, 1)
y1 = np.array([0]*50 + [1]*50, dtype=float)

w, b, losses = 0.0, 0.0, []
for _ in range(1000):
    p = sigmoid(X1.ravel() * w + b)
    loss = -np.mean(y1 * np.log(p + 1e-15) + (1 - y1) * np.log(1 - p + 1e-15))
    losses.append(loss)
    dw = np.mean((p - y1) * X1.ravel())
    db = np.mean(p - y1)
    w -= 0.3 * dw
    b -= 0.3 * db

decision = -b / w
print(f"w={w:.3f}, b={b:.3f}")
print(f"Decision boundary at x = {decision:.3f}  (expected ≈ 0)")
print(f"Final loss: {losses[-1]:.4f}")
```

```output
w=4.127, b=-0.083
Decision boundary at x = 0.020  (expected ≈ 0)
Final loss: 0.1034
```

## Exercise

Implement logistic regression with L2 regularisation: add $\lambda\|w\|_2^2/2$ to the loss and $\lambda w$ to the gradient. Use make_classification with n_features=20, n_informative=5. Compare the test accuracy of unregularised vs regularised ($\lambda=0.1, 1, 10$). Which $\lambda$ works best?

<details>
<summary>Solution</summary>

```python
import numpy as np
from sklearn.datasets import make_classification
from sklearn.preprocessing import StandardScaler
from sklearn.model_selection import train_test_split

def sigmoid(z): return 1 / (1 + np.exp(-z))

X, y = make_classification(n_samples=400, n_features=20, n_informative=5, random_state=0)
X = StandardScaler().fit_transform(X)
X_tr, X_te, y_tr, y_te = train_test_split(X, y, test_size=0.2, random_state=0)

def fit_l2(X, y, lam=0.0, lr=0.1, epochs=1000):
    n, p = X.shape
    w = np.zeros(p); b = 0.0
    for _ in range(epochs):
        ph = sigmoid(X @ w + b)
        dw = (X.T @ (ph - y)) / n + lam * w
        db = np.mean(ph - y)
        w -= lr * dw; b -= lr * db
    return w, b

for lam in [0, 0.1, 1, 10]:
    w, b = fit_l2(X_tr, y_tr, lam=lam)
    acc = np.mean((sigmoid(X_te @ w + b) >= 0.5) == y_te)
    print(f"lambda={lam:4}: test acc={acc:.3f}")
```

The best lambda is typically 0.1 or 1 — enough regularisation to handle the 15 noise features without over-shrinking the 5 informative ones.
</details>
